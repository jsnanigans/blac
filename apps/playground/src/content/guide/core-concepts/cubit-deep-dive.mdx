## The Golden Rule: Immutability

Before diving into advanced patterns, let's reinforce the **most important rule** of state management: **never mutate state directly**.

```typescript
interface AppState {
  user: {
    name: string;
    settings: {
      theme: string;
    };
  };
}

class AppCubit extends Cubit<AppState> {
  // ❌ WRONG: Direct mutation
  changeThemeBad = () => {
    this.state.user.settings.theme = 'dark'; // Mutation!
    this.emit(this.state); // React won't detect the change
  };

  // ✅ CORRECT: Create new objects
  changeTheme = (theme: string) => {
    this.patch({
      user: {
        ...this.state.user,
        settings: {
          ...this.state.user.settings,
          theme,
        },
      },
    });
  };
}
```

React relies on **object reference comparisons** to detect changes. When you mutate an object, the reference stays the same, so React thinks nothing changed. Always create **new objects** when updating state.

<GuideArticleNote title="Why Immutability Matters" className="border-blue-200 bg-blue-50 dark:border-blue-900 dark:bg-blue-950">
Immutability isn't just a React requirement—it enables:
- **Time-travel debugging** - Replay past states
- **Undo/redo functionality** - Keep state history
- **Predictable updates** - No side effects from mutations
- **Performance optimizations** - Fast reference equality checks

Embrace immutability from the start. It'll save you countless debugging hours.
</GuideArticleNote>

## Managing Nested State

Real applications have **complex state structures** with nested objects and arrays. Managing these requires careful spreading to maintain immutability.

### The Spreading Pattern

```typescript
interface UserProfile {
  id: string;
  name: string;
  email: string;
  preferences: {
    theme: 'light' | 'dark';
    notifications: boolean;
    language: string;
  };
  stats: {
    postsCount: number;
    followersCount: number;
  };
}

class UserProfileCubit extends Cubit<UserProfile> {
  constructor() {
    super({
      id: '123',
      name: 'John Doe',
      email: 'john@example.com',
      preferences: {
        theme: 'light',
        notifications: true,
        language: 'en',
      },
      stats: {
        postsCount: 0,
        followersCount: 0,
      },
    });
  }

  // Updating a top-level property (easy)
  updateName = (name: string) => {
    this.patch({ name }); // Simple!
  };

  // Updating a nested property (requires spreading)
  toggleTheme = () => {
    this.patch({
      preferences: {
        ...this.state.preferences, // Keep existing preferences
        theme: this.state.preferences.theme === 'light' ? 'dark' : 'light',
      },
    });
  };

  // Helper for flexible nested updates
  updatePreferences = (preferences: Partial<UserProfile['preferences']>) => {
    this.patch({
      preferences: {
        ...this.state.preferences,
        ...preferences, // Merge new preferences
      },
    });
  };

  // Updating deep nested properties
  incrementPosts = () => {
    this.patch({
      stats: {
        ...this.state.stats, // Keep other stats
        postsCount: this.state.stats.postsCount + 1,
      },
    });
  };
}
```

### Helper Methods

For deeply nested state, create **helper methods** that encapsulate the spreading logic:

```typescript
class UserProfileCubit extends Cubit<UserProfile> {
  // Public API - clean and simple
  enableNotifications = () => {
    this.updatePreferences({ notifications: true });
  };

  // Private helper - handles complexity
  private updatePreferences = (updates: Partial<UserProfile['preferences']>) => {
    this.patch({
      preferences: {
        ...this.state.preferences,
        ...updates,
      },
    });
  };
}
```

This keeps your public API clean while hiding the spreading complexity.

<GuideArticleNote title="When Nesting Gets Too Deep" className="border-amber-200 bg-amber-50 dark:border-amber-900 dark:bg-amber-950">
If you're spreading more than 2-3 levels deep, consider:
1. **Flattening your state** - Reduce nesting levels
2. **Splitting into multiple Cubits** - Separate concerns
3. **Using libraries** - Tools like Immer can help, but understand the basics first

Deep nesting is often a sign of poorly structured state. Refactor before it becomes unmanageable.
</GuideArticleNote>

## Computed Properties with Getters

Sometimes you need values **derived from state** without storing them redundantly. TypeScript getters are perfect for this.

```typescript
interface Todo {
  id: string;
  text: string;
  completed: boolean;
}

interface TodoState {
  todos: Todo[];
  filter: 'all' | 'active' | 'completed';
}

class TodoCubit extends Cubit<TodoState> {
  constructor() {
    super({
      todos: [],
      filter: 'all',
    });
  }

  // Computed property: filtered todos based on current filter
  get filteredTodos(): Todo[] {
    const { todos, filter } = this.state;
    if (filter === 'active') return todos.filter((t) => !t.completed);
    if (filter === 'completed') return todos.filter((t) => t.completed);
    return todos;
  }

  // Computed property: completed count
  get completedCount(): number {
    return this.state.todos.filter((t) => t.completed).length;
  }

  // Computed property: active count
  get activeCount(): number {
    return this.state.todos.filter((t) => !t.completed).length;
  }

  // Computed property: progress percentage
  get progress(): number {
    const total = this.state.todos.length;
    if (total === 0) return 0;
    return Math.round((this.completedCount / total) * 100);
  }

  addTodo = (text: string) => {
    const newTodo: Todo = {
      id: Date.now().toString(),
      text,
      completed: false,
    };
    this.patch({ todos: [...this.state.todos, newTodo] });
  };

  toggleTodo = (id: string) => {
    this.patch({
      todos: this.state.todos.map((todo) =>
        todo.id === id ? { ...todo, completed: !todo.completed } : todo
      ),
    });
  };

  setFilter = (filter: TodoState['filter']) => {
    this.patch({ filter });
  };
}
```

### Using Computed Properties in Components

```tsx
function TodoList() {
  const [state, cubit] = useBloc(TodoCubit);

  return (
    <div>
      {/* Access computed properties from the cubit instance */}
      <p>Progress: {cubit.progress}%</p>
      <p>
        Completed: {cubit.completedCount} / {state.todos.length}
      </p>

      {/* Use computed filtered list */}
      {cubit.filteredTodos.map((todo) => (
        <TodoItem key={todo.id} todo={todo} onToggle={cubit.toggleTodo} />
      ))}
    </div>
  );
}
```

Notice how computed properties are accessed from the **cubit instance**, not the state object.

<GuideArticleNote title="Performance Consideration">
Getters are **recalculated on every access**. For expensive computations:
- Calculate during state updates and store the result
- Use memoization libraries like `memoizee`
- Consider if the computation is actually expensive (profiling first!)

For most use cases, getters are fast enough. Optimize only when profiling shows a problem.
</GuideArticleNote>

## Async Operations Pattern

Handling asynchronous operations (API calls, file uploads, etc.) is one of the most common Cubit use cases. Here's a **reusable pattern**:

```typescript
// Generic async state interface
interface ApiState<T> {
  data: T | null;
  loading: boolean;
  error: string | null;
}

// Generic async Cubit base class
class ApiCubit<T> extends Cubit<ApiState<T>> {
  constructor() {
    super({
      data: null,
      loading: false,
      error: null,
    });
  }

  // Generic fetch method
  protected fetch = async (fetcher: () => Promise<T>) => {
    // Set loading state
    this.emit({ data: null, loading: true, error: null });

    try {
      const data = await fetcher();
      // Success state
      this.emit({ data, loading: false, error: null });
    } catch (error) {
      // Error state
      this.emit({
        data: null,
        loading: false,
        error: error instanceof Error ? error.message : 'An error occurred',
      });
    }
  };

  reset = () => {
    this.emit({ data: null, loading: false, error: null });
  };
}
```

### Extending for Specific Use Cases

```typescript
interface User {
  id: number;
  name: string;
  email: string;
}

class UserApiCubit extends ApiCubit<User[]> {
  fetchUsers = () => {
    return this.fetch(async () => {
      const response = await fetch('/api/users');
      if (!response.ok) {
        throw new Error(`HTTP ${response.status}: ${response.statusText}`);
      }
      return response.json();
    });
  };
}

// Usage in component
function UserList() {
  const [state, api] = useBloc(UserApiCubit);

  return (
    <div>
      <button onClick={api.fetchUsers} disabled={state.loading}>
        {state.loading ? 'Loading...' : 'Fetch Users'}
      </button>

      {state.error && <div className="error">{state.error}</div>}

      {state.data && (
        <ul>
          {state.data.map((user) => (
            <li key={user.id}>{user.name}</li>
          ))}
        </ul>
      )}
    </div>
  );
}
```

### Handling Cancellation

For long-running operations, add cancellation support:

```typescript
class CancellableApiCubit<T> extends Cubit<ApiState<T>> {
  private abortController: AbortController | null = null;

  fetch = async (fetcher: (signal: AbortSignal) => Promise<T>) => {
    // Cancel previous request if still running
    this.abortController?.abort();
    this.abortController = new AbortController();

    this.emit({ data: null, loading: true, error: null });

    try {
      const data = await fetcher(this.abortController.signal);
      this.emit({ data, loading: false, error: null });
    } catch (error) {
      // Don't update state if aborted
      if (error instanceof Error && error.name === 'AbortError') {
        return;
      }
      this.emit({
        data: null,
        loading: false,
        error: error instanceof Error ? error.message : 'An error occurred',
      });
    }
  };

  dispose = () => {
    this.abortController?.abort();
    super.dispose();
  };
}
```

<GuideArticleNote title="Cleanup is Critical" className="border-red-200 bg-red-50 dark:border-red-900 dark:bg-red-950">
Always clean up async operations:
- **Cancel pending requests** when components unmount
- **Ignore state updates** after disposal
- **Clear timers and intervals** in dispose()

Forgetting cleanup leads to memory leaks and race conditions. Override `dispose()` when you have resources to clean up.
</GuideArticleNote>

## Lifecycle and Cleanup

Cubits have a lifecycle that mirrors component lifecycles. Understanding this helps you manage resources properly.

### Lifecycle Stages

```typescript
class ResourceCubit extends Cubit<State> {
  private intervalId: number | null = null;
  private eventSource: EventSource | null = null;

  constructor() {
    super(initialState);

    // Initialize resources
    this.setupPolling();
    this.connectToSSE();
  }

  private setupPolling = () => {
    // Poll every 5 seconds
    this.intervalId = window.setInterval(() => {
      this.refreshData();
    }, 5000);
  };

  private connectToSSE = () => {
    this.eventSource = new EventSource('/api/events');
    this.eventSource.onmessage = (event) => {
      this.handleEvent(JSON.parse(event.data));
    };
  };

  // Override dispose to clean up
  dispose = () => {
    // Clear interval
    if (this.intervalId) {
      clearInterval(this.intervalId);
      this.intervalId = null;
    }

    // Close SSE connection
    if (this.eventSource) {
      this.eventSource.close();
      this.eventSource = null;
    }

    // Call parent dispose
    super.dispose();
  };
}
```

### When to Override Dispose

Override `dispose()` when your Cubit:
- Has timers or intervals (`setTimeout`, `setInterval`)
- Maintains WebSocket or SSE connections
- Subscribes to external event emitters
- Holds expensive resources (file handles, workers, etc.)

<GuideArticleNote title="Automatic vs Manual Disposal">
- **Isolated Cubits** are automatically disposed when components unmount
- **Shared Cubits** must be manually disposed or they'll leak memory
- **Keep-alive Cubits** must always be manually disposed

Know your instance pattern and dispose accordingly!
</GuideArticleNote>

## Testing Cubits

Cubits are easy to test because they're **just classes** with no React dependencies.

```typescript
import { describe, it, expect, beforeEach } from 'vitest';

describe('TodoCubit', () => {
  let cubit: TodoCubit;

  beforeEach(() => {
    cubit = new TodoCubit();
  });

  it('should start with empty todos', () => {
    expect(cubit.state.todos).toEqual([]);
    expect(cubit.state.filter).toBe('all');
  });

  it('should add a todo', () => {
    cubit.addTodo('Learn BlaC');

    expect(cubit.state.todos).toHaveLength(1);
    expect(cubit.state.todos[0].text).toBe('Learn BlaC');
    expect(cubit.state.todos[0].completed).toBe(false);
  });

  it('should toggle a todo', () => {
    cubit.addTodo('Test task');
    const todoId = cubit.state.todos[0].id;

    cubit.toggleTodo(todoId);
    expect(cubit.state.todos[0].completed).toBe(true);

    cubit.toggleTodo(todoId);
    expect(cubit.state.todos[0].completed).toBe(false);
  });

  it('should calculate progress correctly', () => {
    cubit.addTodo('Task 1');
    cubit.addTodo('Task 2');
    cubit.addTodo('Task 3');

    expect(cubit.progress).toBe(0);

    cubit.toggleTodo(cubit.state.todos[0].id);
    expect(cubit.progress).toBe(33); // 1/3 = 33%

    cubit.toggleTodo(cubit.state.todos[1].id);
    expect(cubit.progress).toBe(67); // 2/3 = 67%

    cubit.toggleTodo(cubit.state.todos[2].id);
    expect(cubit.progress).toBe(100); // 3/3 = 100%
  });

  it('should filter active todos', () => {
    cubit.addTodo('Active 1');
    cubit.addTodo('Completed 1');
    cubit.toggleTodo(cubit.state.todos[1].id);
    cubit.addTodo('Active 2');

    cubit.setFilter('active');
    expect(cubit.filteredTodos).toHaveLength(2);
    expect(cubit.activeCount).toBe(2);
  });

  it('should filter completed todos', () => {
    cubit.addTodo('Task 1');
    cubit.addTodo('Task 2');
    cubit.toggleTodo(cubit.state.todos[0].id);

    cubit.setFilter('completed');
    expect(cubit.filteredTodos).toHaveLength(1);
    expect(cubit.completedCount).toBe(1);
  });
});
```

## Performance Optimization

### 1. Batch State Updates

When making multiple state changes, batch them:

```typescript
// ❌ Multiple updates (triggers multiple re-renders)
updateUser = (name: string, email: string) => {
  this.patch({ name });
  this.patch({ email });
};

// ✅ Single update (one re-render)
updateUser = (name: string, email: string) => {
  this.patch({ name, email });
};
```

### 2. Avoid Unnecessary Computed Properties

```typescript
// ❌ Recomputed on every render
get expensiveCalculation(): number {
  return this.state.items.reduce((sum, item) => {
    // Complex calculation
    return sum + item.value * Math.sqrt(item.weight);
  }, 0);
}

// ✅ Calculate during updates, store result
updateItems = (items: Item[]) => {
  const total = items.reduce((sum, item) => {
    return sum + item.value * Math.sqrt(item.weight);
  }, 0);

  this.patch({ items, cachedTotal: total });
};
```

### 3. Use Selectors in Components

BlaC's dependency tracking optimizes re-renders based on what fields components actually use:

```tsx
// Only re-renders when count changes
function Counter() {
  const [state] = useBloc(AppCubit);
  return <div>{state.count}</div>;
}

// Only re-renders when user.name changes
function UserName() {
  const [state] = useBloc(AppCubit);
  return <div>{state.user.name}</div>;
}
```

This works automatically—no manual selectors needed!

## Key Takeaways

- **Never mutate state** - Always create new objects with spreading

- **Use getters for computed values** - Derive data instead of storing it
